\begin{frame}{Blocos de construção}
    O modelo de Computação em Nuvem é composto de um front-end e um back-end, conectados por meio de uma rede, geralmente a Internet

    \uncover<2->{
        \begin{block}{Front-End}
            \begin{itemise}
                \item o veículo pelo qual o usuário interage com o sistema
                \item composto de um cliente de computador ou uma rede de
                computadores de um empreendimento
                \item possui os aplicativos usados para acessar a nuvem
            \end{itemise}
        \end{block}
    }
    
    \uncover<3->{
        \begin{block}{Back-End}
            \begin{itemise}
                \item é a própria nuvem de serviços
                \item contém aplicativos, computadores, servidores e armazenamento de
                dados
            \end{itemise}
        \end{block}
    }

\end{frame}
