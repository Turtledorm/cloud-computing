\begin{frame}{Blocos de construção}
    Modelo composto por um front-end e um back-end
    conectados por meio de uma rede, geralmente a Internet

    \uncover<2->{
        \begin{block}{Front-end}
            \begin{itemise}
                \item Veículo pelo qual o usuário interage com o sistema
                \item Composto de um cliente de computador ou uma rede de
                    computadores de um empreendimento
                \item Possui os aplicativos usados para acessar a nuvem
            \end{itemise}
        \end{block}
    }
    
    \uncover<3->{
        \begin{block}{Back-end}
            \begin{itemise}
                \item A própria nuvem de serviços
                \item Contém aplicativos, computadores, servidores e armazenamento de
                    dados
            \end{itemise}
        \end{block}
    }

\end{frame}
