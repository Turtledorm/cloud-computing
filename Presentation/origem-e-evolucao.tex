\begin{frame}{Origem e evolução}

    \uncover<2->{
        \begin{block}{John McCarthy (década de 1960)}
            Apresentou a ideia de computação por tempo compartilhado,
            em que um computador pode ser utilizado simultaneamente por dois ou mais
            usuários para a realização de determinadas tarefas
        \end{block}
    }

    \uncover<3->{
        \begin{block}{Joseph Carl Robnett Licklider (década de 1960)}
            Entendeu que os computadores poderiam ser usados de maneira conectada,
            de forma a permitir comunicação de maneira global e, consequentemente,
            com o compartilhamento de dados
        \end{block}
    }

    \uncover<4->{
        \begin{block}{Eric Schmidt ($\sim$2006)}
            Popularizou o termo \emph{Computação em Nuvem} em uma palestra, descrevendo
            como sua empresa gerenciava seus próprios data centers
        \end{block}
    }
\end{frame}
